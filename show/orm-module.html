ORM 组件职责

SqlSession

职责：数据库会话门面，提供统一的数据库操作接口（selectOne, insert, update, delete）。

类比：类似于银行的柜台窗口，客户通过窗口办理各种业务。

Executor

职责：SQL 执行器，负责 SQL 的实际执行，包括查询、更新、事务管理等。

实现：BaseExecutor 使用模板方法模式定义执行流程，SimpleExecutor 是简单实现。

StatementHandler

职责：SQL 语句处理器，负责创建 PreparedStatement、设置参数、执行查询。

为什么需要：封装 JDBC 的 Statement 操作，让代码更清晰。

LightSSM 源码实现

1. Executor 模板方法

// 模板方法：定义查询流程 class BaseExecutor { query(ms, parameter, rowBounds) { // 1. 获取 BoundSql boundSql = ms.getBoundSql(parameter) // 2. 创建 CacheKey（一级缓存） key = createCacheKey(ms, parameter, rowBounds, boundSql) // 3. 查询本地缓存 cachedResult = localCache.get(key) if cachedResult != null: return cachedResult // 4. 执行数据库查询（子类实现） result = doQuery(ms, parameter, rowBounds, boundSql) // 5. 缓存结果 localCache.put(key, result) return result } // 抽象方法：由子类实现具体查询 abstract doQuery(ms, parameter, rowBounds, boundSql) } // 简单执行器实现 class SimpleExecutor extends BaseExecutor { doQuery(ms, parameter, rowBounds, boundSql) { result = new List() conn = transaction.getConnection() // 1. 创建 StatementHandler sh = newStatementHandler(ms, boundSql) // 2. 准备 PreparedStatement stmt = sh.prepare(conn) // 3. 设置参数 sh.parameterize(stmt) // 4. 执行查询 result = sh.query(stmt) return result } }

2. Mapper 代理机制

// Mapper 代理：动态代理实现 class MapperProxy implements InvocationHandler { sqlSession = SqlSession mapperInterface = Class methodCache = Map<Method, MapperMethod> invoke(proxy, method, args) { // Object 方法直接调用 if method.declaringClass == Object.class: return method.invoke(this, args) // 缓存 MapperMethod mapperMethod = methodCache.computeIfAbsent(method) // 执行 SQL return mapperMethod.execute(sqlSession, args) } } // MapperMethod 执行 SQL class MapperMethod { execute(sqlSession, args) { switch (command.type) { case SELECT: if method.returnsList(): return sqlSession.selectList(command.name, args) else: return sqlSession.selectOne(command.name, args) case INSERT: return sqlSession.insert(command.name, args) case UPDATE: return sqlSession.update(command.name, args) case DELETE: return sqlSession.delete(command.name, args) } } }

3. 结果集映射

// 结果集处理器：ResultSet -> Java 对象 class DefaultResultSetHandler { handleResultSets(stmt) { result = new List() rs = stmt.getResultSet() while rs.next(): // 1. 创建结果对象 resultObject = createResultObject(rs) // 2. 自动映射列到属性 applyAutomaticMappings(resultObject, rs) // 3. 显式映射（如果有 ResultMap） applyPropertyMappings(resultObject, rs) result.add(resultObject) return result } // 自动映射：列名转驼峰 applyAutomaticMappings(obj, rs) { meta = rs.getMetaData() for i in 1..meta.getColumnCount(): columnName = meta.getColumnName(i) propertyName = camelCase(columnName) // 获取 TypeHandler th = typeHandlerRegistry.getTypeHandler(rs.getObject(columnName).class) // 获取值并设置 value = th.getResult(rs, columnName) setProperty(obj, propertyName, value) } }
