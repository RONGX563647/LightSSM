1. 类型系统架构

1.1 TypeHandler接口

LightSSM实现：

public interface TypeHandler {

    T getResult(ResultSet rs, String columnName) throws SQLException;

public interface TypeHandler {

2. 基础类型处理器

2.1 BaseTypeHandler基类

LightSSM实现：

public abstract class BaseTypeHandler implements TypeHandler {
    
    @Override
    public void setParameter(PreparedStatement ps, int i, T parameter, JdbcType jdbcType) throws SQLException {
        if (parameter == null) {
            if (jdbcType == null) {
                throw new TypeException("JDBC type is null");
            }
            ps.setNull(i, jdbcType.TYPE_CODE);
        } else {
            setNonNullParameter(ps, i, parameter, jdbcType);
        }
    }
    
    protected abstract void setNonNullParameter(PreparedStatement ps, int i, T parameter, JdbcType jdbcType) throws SQLException;
}

2.2 StringTypeHandler

LightSSM实现：

public class StringTypeHandler extends BaseTypeHandler {
    
    @Override
    protected void setNonNullParameter(PreparedStatement ps, int i, String parameter, JdbcType jdbcType) throws SQLException {
        ps.setString(i, parameter);
    }
    
    @Override
    public String getResult(ResultSet rs, String columnName) throws SQLException {
        return rs.getString(columnName);
    }
    
    @Override
    public String getResult(ResultSet rs, int columnIndex) throws SQLException {
        return rs.getString(columnIndex);
    }
}

3. 类型处理器注册表

3.1 TypeHandlerRegistry

LightSSM实现：

public class TypeHandlerRegistry {
    private final Map> jdbcTypeHandlerMap = new HashMap<>();
    private final Map> typeHandlerStringMap = new HashMap<>();
    private final Map, TypeHandler> typeHandlerClassMap = new HashMap<>();
    
    public TypeHandlerRegistry() {
        // 注册内置类型处理器
        register(String.class, new StringTypeHandler());
        register(Integer.class, new IntegerTypeHandler());
        register(int.class, new IntegerTypeHandler());
        register(Long.class, new LongTypeHandler());
        register(long.class, new LongTypeHandler());
        register(Double.class, new DoubleTypeHandler());
        register(double.class, new DoubleTypeHandler());
        register(Boolean.class, new BooleanTypeHandler());
        register(boolean.class, new BooleanTypeHandler());
        register(Date.class, new DateTypeHandler());
        // ... 更多注册
    }
    
    @SuppressWarnings("unchecked")
    public  void register(Class javaTypeClass, TypeHandler handler) {
        if (javaTypeClass != null) {
            typeHandlerClassMap.put(javaTypeClass, handler);
        }
    }
    
    @SuppressWarnings("unchecked")
    public  TypeHandler getTypeHandler(Class type) {
        TypeHandler handler = typeHandlerClassMap.get(type);
        if (handler == null) {
            // 查找父类
            for (Map.Entry, TypeHandler> entry : typeHandlerClassMap.entrySet()) {
                if (entry.getKey().isAssignableFrom(type)) {
                    return (TypeHandler) entry.getValue();
                }
            }
            // 默认返回Object类型处理器
            handler = typeHandlerClassMap.get(Object.class);
        }
        return (TypeHandler) handler;
    }
}

4. 类型别名系统

4.1 TypeAliasRegistry

LightSSM实现：

public class TypeAliasRegistry {
    private final Map> typeAliases = new HashMap<>();
    
    public TypeAliasRegistry() {
        // 注册内置类型别名
        registerAlias("string", String.class);
        registerAlias("int", Integer.class);
        registerAlias("integer", Integer.class);
        registerAlias("long", Long.class);
        registerAlias("double", Double.class);
        registerAlias("boolean", Boolean.class);
        registerAlias("date", Date.class);
        registerAlias("map", Map.class);
        registerAlias("list", List.class);
    }
    
    public void registerAlias(String alias, Class value) {
        if (alias == null) {
            throw new TypeException("Alias cannot be null");
        }
        typeAliases.put(alias.toLowerCase(Locale.ENGLISH), value);
    }
    
    public void registerAlias(String className) {
        try {
            registerAlias(className, Resources.classForName(className));
        } catch (ClassNotFoundException e) {
            throw new TypeException("Error registering type alias: " + className, e);
        }
    }
    
    public  Class resolveAlias(String string) {
        if (string == null) return null;
        String key = string.toLowerCase(Locale.ENGLISH);
        Class value = typeAliases.get(key);
        if (value == null) {
            try {
                return (Class) Resources.classForName(string);
            } catch (ClassNotFoundException e) {
                throw new TypeException("Could not resolve type alias: " + string, e);
            }
        }
        return (Class) value;
    }
}

6. 类型转换场景

6.1 参数绑定

// PreparedStatement参数绑定
public class DefaultParameterHandler implements ParameterHandler {
    @Override
    public void setParameters(PreparedStatement ps) throws SQLException {
        BoundSql boundSql = mappedStatement.getBoundSql();
        List parameterMappings = boundSql.getParameterMappings();
        
        for (int i = 0; i < parameterMappings.size(); i++) {
            ParameterMapping parameterMapping = parameterMappings.get(i);
            Object value = boundSql.getParameterObject();
            JdbcType jdbcType = parameterMapping.getJdbcType();
            
            // 获取TypeHandler
            TypeHandler typeHandler = parameterMapping.getTypeHandler();
            
            // 设置参数
            ((TypeHandler) typeHandler).setParameter(ps, i + 1, value, jdbcType);
        }
    }
}
