```java
interface SqlSession {
    + selectOne(statement, parameter): T
    + selectList(statement, parameter): List<E>
    + insert/statement, parameter): int
    + update/statement, parameter): int
    + delete/statement, parameter): int
    + commit()
    + getMapper(type): T
    + getConfiguration(): Configuration
}

interface SqlSessionFactory {
    + openSession(): SqlSession
    + openSession(autoCommit: boolean): SqlSession
}

class DefaultSqlSession implements SqlSession { }
class DefaultSqlSessionFactory implements SqlSessionFactory { }
```

### 4.2 Executorе®¶ж—Ҹ

```java
interface Executor {
    + update(ms, parameter): int
    + query(ms, parameter, rowBounds, resultHandler): List<E>
    + commit(required)
    + rollback(required)
    + close(forceRollback)
}

abstract class BaseExecutor implements Executor {
    + query(ms, parameter, rowBounds, resultHandler): List<E>  // жЁЎжқҝж–№жі•
    # doQuery(ms, parameter, rowBounds, resultHandler, boundSql): List<E>  // жҠҪиұЎж–№жі•
    # doUpdate(ms, parameter): int  // жҠҪиұЎж–№жі•
}

class SimpleExecutor extends BaseExecutor { }
```

### 4.3 StatementHandlerе®¶ж—Ҹ

```java
interface StatementHandler {
    + prepare(connection): Statement

    + update(statement): int
    + query(statement, resultHandler): List<E>

abstract class BaseStatementHandler implements StatementHandler { }

class PreparedStatementHandler extends BaseStatementHandler { }
```
