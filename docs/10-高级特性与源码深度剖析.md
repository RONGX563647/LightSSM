# 10 - 高级特性与源码深度剖析

## 1. 反射系统

### 1.1 Reflector反射器

**LightSSM实现**：
```java
public class Reflector {
    private final Class<?> type;
    private final String[] readablePropertyNames;
    private final String[] writablePropertyNames;
    private final Map<String, Method> getMethods = new HashMap<>();
    private final Map<String, Method> setMethods = new HashMap<>();
    private final Map<String, Class<?>> getTypes = new HashMap<>();
    private final Map<String, Class<?>> setTypes = new HashMap<>();
    
    public Reflector(Class<?> type) {
        this.type = type;
        addDefaultConstructor(type);
        addGetMethods(type);
        addSetMethods(type);
        addFields(type);
        
        this.readablePropertyNames = getMethods.keySet().toArray(new String[0]);
        this.writablePropertyNames = setMethods.keySet().toArray(new String[0]);
    }
    
    private void addGetMethods(Class<?> type) {
        Method[] methods = type.getMethods();
        for (Method method : methods) {
            String name = method.getName();
            if (name.startsWith("get") && name.length() > 3 
                && method.getParameterCount() == 0) {
                String propertyName = method.getName().substring(3);
                propertyName = propertyName.substring(0, 1).toLowerCase() 
                    + propertyName.substring(1);
                getMethods.put(propertyName, method);
                getTypes.put(propertyName, method.getReturnType());
            }
        }
    }
    
    private void addSetMethods(Class<?> type) {
        Method[] methods = type.getMethods();
        for (Method method : methods) {
            String name = method.getName();
            if (name.startsWith("set") && name.length() > 3 
                && method.getParameterCount() == 1) {
                String propertyName = method.getName().substring(3);
                propertyName = propertyName.substring(0, 1).toLowerCase() 
                    + propertyName.substring(1);
                setMethods.put(propertyName, method);
                setTypes.put(propertyName, method.getParameterTypes()[0]);
            }
        }
    }
    
    public Method getGetMethod(String propertyName) {
        return getMethods.get(propertyName);
    }
    
    public Method getSetMethod(String propertyName) {
        return setMethods.get(propertyName);
    }
    
    public Class<?> getGetterType(String propertyName) {
        return getTypes.get(propertyName);
    }
    
    public Class<?> getSetterType(String propertyName) {
        return setTypes.get(propertyName);
    }
}
```

### 1.2 与MyBatis官方对比

**MyBatis官方的Reflector**：
- 支持字段直接访问（无getter/setter）
- 支持大小写不敏感属性查找
- 支持缓存方法查找结果
- 支持私有构造函数检测
- 支持协变返回类型

**LightSSM简化**：
- 基础的getter/setter查找
- 不支持字段访问
- 不支持大小写不敏感

## 2. MetaObject元对象

### 2.1 MetaObject实现

**LightSSM实现**：
```java
public class MetaObject {
    private final Object originalObject;
    private final ReflectorFactory reflectorFactory;
    private final ObjectWrapperFactory objectWrapperFactory;
    private final ObjectWrapper objectWrapper;
    
    private MetaObject(Object object, ReflectorFactory reflectorFactory, 
                       ObjectWrapperFactory objectWrapperFactory) {
        this.originalObject = object;
        this.reflectorFactory = reflectorFactory;
        this.objectWrapperFactory = objectWrapperFactory;
        
        if (object instanceof ObjectWrapper) {
            this.objectWrapper = (ObjectWrapper) object;
        } else if (objectWrapperFactory.hasWrapperFor(object)) {
            this.objectWrapper = objectWrapperFactory.getWrapperFor(this, object);
        } else {
            this.objectWrapper = new BeanWrapper(this, object);
        }
    }
    
    public static MetaObject forObject(Object object, ObjectFactory objectFactory, 
                                       ObjectWrapperFactory objectWrapperFactory) {
        if (object == null) {
            return SystemMetaObject.NULL_META_OBJECT;
        }
        return new MetaObject(object, new DefaultReflectorFactory(), 
            objectWrapperFactory);
    }
    
    public String findProperty(String name, boolean useCamelCaseMapping) {
        return objectWrapper.findProperty(name, useCamelCaseMapping);
    }
    
    public Object getValue(String name) {
        PropertyTokenizer prop = new PropertyTokenizer(name);
        if (prop.hasNext()) {
            MetaObject metaValue = metaObjectForProperty(prop.getName());
            if (metaValue == null || metaValue.getOriginalObject() == null) {
                return null;
            }
            return metaValue.getValue(prop.getChildren());
        } else {
            return objectWrapper.get(prop);
        }
    }
    
    public void setValue(String name, Object value) {
        PropertyTokenizer prop = new PropertyTokenizer(name);
        if (prop.hasNext()) {
            MetaObject metaValue = metaObjectForProperty(prop.getName());
            if (metaValue == null) {
                throw new RuntimeException("Property '" + prop.getName() + "' is null");
            }
            metaValue.setValue(prop.getChildren(), value);
        } else {
            objectWrapper.set(prop, value);
        }
    }
    
    private MetaObject metaObjectForProperty(String name) {
        Object obj = objectWrapper.get(new PropertyTokenizer(name));
        return MetaObject.forObject(obj, null, objectWrapperFactory);
    }
}
```

### 2.2 PropertyTokenizer属性解析器

```java
public class PropertyTokenizer implements Iterable<PropertyTokenizer> {
    private String name;           // 当前属性名
    private String indexedName;    // 带索引的属性名
    private String index;          // 索引值
    private String children;       // 子属性
    
    public PropertyTokenizer(String fullname) {
        int delim = fullname.indexOf('.');
        if (delim > -1) {
            name = fullname.substring(0, delim);
            children = fullname.substring(delim + 1);
        } else {
            name = fullname;
            children = null;
        }
        indexedName = name;
        delim = name.indexOf('[');
        if (delim > -1) {
            index = name.substring(delim + 1, name.length() - 1);
            name = name.substring(0, delim);
        }
    }
    
    public String getName() {
        return name;
    }
    
    public String getIndex() {
        return index;
    }
    
    public String getChildren() {
        return children;
    }
    
    public boolean hasNext() {
        return children != null;
    }
}
```

### 2.3 与MyBatis官方对比

**MyBatis官方的MetaObject**：
- 支持嵌套属性（user.address.city）
- 支持集合索引（list[0].name）
- 支持Map键访问（map.key）
- 支持对象包装器（ObjectWrapper）
- 支持空对象处理（NULL_META_OBJECT）

**LightSSM简化**：
- 基础的嵌套属性支持
- 不支持集合索引
- 简化对象包装器

### 3.2 DefaultObjectFactory实现

```java
public class DefaultObjectFactory implements ObjectFactory {
    @Override
    public <T> T create(Class<T> type) {
        try {
            return type.getDeclaredConstructor().newInstance();
        } catch (Exception e) {
            throw new RuntimeException("Failed to create object: " + type.getName(), e);
        }
    }
    
    @Override
    public <T> T create(Class<T> type, List<Class<?>> constructorArgTypes, 
                       List<Object> constructorArgs) {
        try {
            Constructor<?> constructor = type.getDeclaredConstructor(
                constructorArgTypes.toArray(new Class[0]));
            return (T) constructor.newInstance(constructorArgs.toArray(new Object[0]));
        } catch (Exception e) {
            throw new RuntimeException("Failed to create object: " + type.getName(), e);
        }
    }
    
    @Override
    public void setProperties(Properties properties) {
        // 可以配置工厂行为
    }
}
```

### 4.2 BeanWrapper实现

```java
public class BeanWrapper implements ObjectWrapper {
    private final Object object;
    private final Reflector reflector;
    
    public BeanWrapper(MetaObject metaObject, Object object) {
        this.object = object;
        this.reflector = new Reflector(object.getClass());
    }
    
    @Override
    public Object get(PropertyTokenizer prop) {
        Method method = reflector.getGetMethod(prop.getName());
        if (method == null) {
            throw new RuntimeException("No getter for '" + prop.getName() + "'");
        }
        try {
            return method.invoke(object);
        } catch (Exception e) {
            throw new RuntimeException("Error getting '" + prop.getName() + "'", e);
        }
    }
    
    @Override
    public void set(PropertyTokenizer prop, Object value) {
        Method method = reflector.getSetMethod(prop.getName());
        if (method == null) {
            throw new RuntimeException("No setter for '" + prop.getName() + "'");
        }
        try {
            method.invoke(object, value);
        } catch (Exception e) {
            throw new RuntimeException("Error setting '" + prop.getName() + "'", e);
        }
    }
    
    @Override
    public String findProperty(String name, boolean useCamelCaseMapping) {
        if (reflector.getGetMethod(name) != null || reflector.getSetMethod(name) != null) {
            return name;
        }
        return null;
    }
}
```

## 5. 缓存系统

### 5.1 Cache接口

**MyBatis官方Cache接口**：
```java
public interface Cache {
    String getId();
    void putObject(Object key, Object value);
    Object getObject(Object key);
    Object removeObject(Object key);
    void clear();
    int getSize();
}
```

**LightSSM简化实现**：
```java
public class PerpetualCache implements Cache {
    private final String id;
    private final Map<Object, Object> cache = new HashMap<>();
    
    public PerpetualCache(String id) {
        this.id = id;
    }
    
    @Override
    public String getId() {
        return id;
    }
    
    @Override
    public void putObject(Object key, Object value) {
        cache.put(key, value);
    }
    
    @Override
    public Object getObject(Object key) {
        return cache.get(key);
    }
    
    @Override
    public Object removeObject(Object key) {
        return cache.remove(key);
    }
    
    @Override
    public void clear() {
        cache.clear();
    }
    
    @Override
    public int getSize() {
        return cache.size();
    }
}
```

### 5.2 缓存装饰器

**MyBatis官方的缓存装饰器链**：
```
PerpetualCache（基础缓存）
├── LruCache（LRU淘汰）
├── FifoCache（FIFO淘汰）
├── SoftCache（软引用）
├── WeakCache（弱引用）
├── SerializedCache（序列化）
├── SynchronizedCache（同步）
├── ScheduledCache（定时清空）
└── LoggingCache（日志）
```

**LightSSM**：仅实现基础的PerpetualCache。

public class ParameterMapping {
    private String property;

    private ParameterMapping(Builder builder) {
        this.property = builder.property;

        this.jdbcType = builder.jdbcType;
        this.typeHandler = builder.typeHandler;
        this.mode = builder.mode;
    }

        return javaType;
    }
    
    public JdbcType getJdbcType() {
        return jdbcType;
    }
    
    public TypeHandler<?> getTypeHandler() {
        return typeHandler;
    }
    
    public static class Builder {
        private String property;
        private Class<?> javaType;
        private JdbcType jdbcType;
        private TypeHandler<?> typeHandler;
        private String mode = "IN";
        
        public Builder(String property) {
            this.property = property;
        }

            return this;
        }
